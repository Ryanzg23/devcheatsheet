HTTP Status

Redirect all to HTTPS & HTTPS WWW

RewriteEngine On

# Redirect all HTTP requests to HTTPS, preserving www or non-www
RewriteCond %{HTTPS} off
RewriteRule ^ https://%{HTTP_HOST}%{REQUEST_URI} [L,R=301]

Redirect all to HTTPS

RewriteEngine On
RewriteBase /

# Redirect www to non-www (works for both HTTP and HTTPS)
RewriteCond %{HTTP_HOST} ^www\.(.+)$ [NC]
RewriteRule ^ https://%1%{REQUEST_URI} [R=301,L]

# Redirect HTTP to HTTPS for non-www
RewriteCond %{HTTPS} off
RewriteCond %{HTTP_HOST} !^www\. [NC]
RewriteRule ^ https://%{HTTP_HOST}%{REQUEST_URI} [R=301,L]

Redirect all to HTTPS WWW

RewriteEngine On
RewriteCond %{HTTPS} off [OR]
RewriteCond %{HTTP_HOST} !^www\. [NC]
RewriteCond %{HTTP_HOST} ^(?:www\.)?(.+)$ [NC]
RewriteRule ^ https://www.%1%{REQUEST_URI} [L,NE,R=301]

Redirect all to HTTP

RewriteEngine On
RewriteBase /

# Redirect www to non-www (any domain)
RewriteCond %{HTTP_HOST} ^www\.(.+)$ [NC]
RewriteRule ^ http://%1%{REQUEST_URI} [R=301,L]

# Redirect HTTPS to HTTP (only for non-www)
RewriteCond %{HTTPS} on
RewriteCond %{HTTP_HOST} !^www\. [NC]
RewriteRule ^ http://%{HTTP_HOST}%{REQUEST_URI} [R=301,L]

Redirect all to HTTP WWW

<IfModule mod_rewrite.c>
RewriteEngine On
RewriteCond %{HTTPS} on [OR]
RewriteCond %{HTTP_HOST} !^www\. [NC]
RewriteCond %{HTTP_HOST} ^(?:www\.)?(.+)$ [NC]
RewriteRule ^ http://www.%1%{REQUEST_URI} [L,NE,R=301]
</IfModule>

Redirect to new domain

RewriteEngine On
RewriteCond %{HTTP_HOST} ^(www\.)?usergacorvip\.com$ [NC]
RewriteRule ^ https://www.wearepetaloso.com/? [L,R=301]

Allow direct access to this folder's files

Options +FollowSymLinks
RewriteEngine On

# Stop parent .htaccess from interfering
RewriteOptions InheritDownBefore

# Serve index.html automatically
DirectoryIndex index.html

# Don't rewrite anything here
RewriteRule .* - [L]

Redirect index.html to slash

RewriteRule ^(.*/)?index\.html$ /$1 [R=301,L]
